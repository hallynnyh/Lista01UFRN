11. Suponha que as seguintes declarações tenham sido realizadas:
--
float aloha[10], coisas[10][5], *pf, value = 2.2;
int i=3;
--

Identifique quais dos seguintes comandos é válido ou inválido:
a. aloha[2] = value;		// Válido
b. scanf("%f", &aloha); 	// Válido! OBS: Caso não haja preenchimento do vetor, valores estranhos podem aparecer na saída. 
c. aloha = "value";		// Inválido
d. printf("%f", aloha);		// Válido
e. coisas[4][4] = aloha[3];	// Válido
f. coisas[5] = aloha;		// Inválido
g. pf = value;			// Inválido
h. pf = aloha;			// Válido
